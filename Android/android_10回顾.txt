xml解析
  三个文件夹下的xml解析 res_xml  res_raw assets

res_xml
  XmlResourceParser xmlp = getResource().getXml(R.xml.xxx);
  XmlPresourceParser是XmlPullParser的子接口 ，所以XmlPresourceParser实际上是一个pull解析器

res_raw
  //获得解析工厂
  XmlPullParserFactory factory = XmlPullFactory.newInstence();
 //获得解析器
  XmlPullParser p = factory.newPullPaser();
//获得输入流
  InputStream in = getResource().openRawResource(R.raw.xxx);
//设置输入
  p.setInput(in);
