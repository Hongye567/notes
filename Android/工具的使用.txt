自定义组合组件
    编写自定义组件类 MyView 继承 一个布局(LinearLayout)
      重写父类的构造方法
       MyView(Context c)
       {
           this(c,null); //必须在方法体的第一行
       }
       MyView(Context c ,Attribute a)
       {
              //调用父类的构造方法
              super(c,a)
              //初始化控件 
                 Button bt = new Button(getContext()) ;
              //为控件设置属性
                 bt.setText("点击");
              // 为自定义布局添加控件
                 addView(bt);
                 addView(bt,new LinearLayout.LayoutParameters(int width,int hight,float weight));


      }
